23double touched_cells(size_t &ke, size_t &ivend, array_1d_t<real_t> &t,
24 array_1d_t<real_t> &rho, array_1d_t<real_t> &qv,
25 array_1d_t<real_t> &qc, array_1d_t<real_t> &qi,
26 array_1d_t<real_t> &qr, array_1d_t<real_t> &qs,
27 array_1d_t<real_t> &qg) {
28 size_t oned_vec_index;
29 size_t jmx = 0;
30 for (size_t i = ke - 1; i < ke; --i) {
31 for (size_t j = 0; j < ivend; j++) {
32 oned_vec_index = i * ivend + j;
33 if ((std::max({qc[oned_vec_index], qr[oned_vec_index], qs[oned_vec_index],
34 qi[oned_vec_index], qg[oned_vec_index]}) >
35 graupel_ct::qmin) or
36 ((t[oned_vec_index] < graupel_ct::tfrz_het2) and
37 (qv[oned_vec_index] >
38 thermo::qsat_ice_rho(t[oned_vec_index], rho[oned_vec_index])))) {
39 jmx = jmx + 1;
40 }
41 }
42 }
43
44 return static_cast<double>(jmx) / (ke * ivend);
45}
