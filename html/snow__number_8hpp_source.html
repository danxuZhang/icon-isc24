52TARGET real_t snow_number(real_t t, real_t rho, real_t qs) {
53
54 if (qs > graupel_ct::qmin) {
55 real_t tc = fmax(fmin(t, tmax), tmin) - thermodyn::tmelt;
56 real_t alf = pow(static_cast<real_t>(10.), (xa1 + tc * (xa2 + tc * xa3)));
57 real_t bet = xb1 + tc * (xb2 + tc * xb3);
58 real_t n0s =
59 n0s3 *
60 pow(((qs + qsmin) * rho / graupel_ct::ams),
61 (static_cast<real_t>(4.0) - static_cast<real_t>(3) * bet)) /
62 (alf * alf * alf);
63 real_t y = exp(n0s2 * tc);
64 real_t n0smn = fmax(n0s4 * y, n0s5);
65 real_t n0smx = fmin(n0s6 * y, n0s7);
66 return fmin(n0smx, fmax(n0smn, n0s));
67 } else {
68 return n0s0;
69 }
70}
