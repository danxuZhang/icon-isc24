49TARGET real_t vapor_x_graupel(real_t t, real_t p, real_t rho, real_t qg,
50 real_t dvsw, real_t dvsi, real_t dvsw0,
51 real_t dt) {
52
53 real_t result = 0.0;
54
55 if (qg > graupel_ct::qmin) {
56 if (t < thermodyn::tmelt) {
57 result =
58 (a1_vg + a2_vg * t + a3 / p + a4 * p) * dvsi * pow(qg * rho, b_vg);
59 } else {
60 if (t > (thermodyn::tmelt - graupel_ct::tx * dvsw0)) {
61 result = (a5 + a6 * p) * fmin(static_cast<real_t>(0.0), dvsw0) *
62 pow(qg * rho, b_vg);
63 } else {
64 result = (a7 + a8 * p) * dvsw * pow(qg * rho, b_vg);
65 }
66 }
67 result = fmax(result, -qg / dt);
68 }
