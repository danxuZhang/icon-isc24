12// In numerical weather prediction (NWP) models like the one you're asking about,
13// the atmosphere is divided into a three-dimensional grid. Each grid point or "cell"
14// represents a discrete volume of air with its own set of atmospheric properties
15// (such as temperature, pressure, humidity) and, in this context, hydrometeor content.
16// Calculations are performed for each cell, considering interactions with neighboring cells,
17// to simulate the dynamics of the atmosphere over time. The term "cell" is commonly used
18// to denote these discrete elements of the computational grid.

76inline auto get_min_element(T* arr, size_t sz) -> T{
77 T min_ele = arr[0];
78 for (size_t i = 0; i < sz; ++i) {
79 if (arr[i] < min_ele) {
80 min_ele = arr[i];
81 }
82 }
83 return min_ele;

110void precip(const real_t (&params)[3], real_t (&precip)[3], real_t zeta,
111 real_t vc, real_t flx, real_t vt, real_t q, real_t q_kp1,
112 real_t rho) {
113 real_t rho_x, flx_eff, flx_partial;
114 // Calculation of the effective density of precipitation in the cell
115 rho_x = q * rho;
116 // Effective flux of precipitation into the cell, considering both the density and the initial flux
117 flx_eff = (rho_x / zeta) + static_cast<real_t>(2.0) * flx;
118 // Partial flux of precipitation, adjusted for the fall speed correction factor
119 flx_partial = rho_x * vc * fall_speed(rho_x, params);
120 flx_partial = std::fmin(flx_partial, flx_eff);
121 // Update the specific mass of hydrometeors in the cell
122 precip[0] = (zeta * (flx_eff - flx_partial)) /
123 ((static_cast<real_t>(1.0) + zeta * vt) * rho);
124 // Update the flux of precipitation from the cell above
125 precip[1] = (precip[0] * rho * vt + flx_partial) * static_cast<real_t>(0.5);
126 // Recalculate the density of precipitation, considering the updates
127 rho_x = (precip[0] + q_kp1) * static_cast<real_t>(0.5) * rho;
128 // Update the terminal velocity of precipitation particles
129 precip[2] = vc * fall_speed(rho_x, params);
130}

534 if (k == kstart) {

541 if (k >= get_min_element(kmin[iv], np)) {

543 qliq = q[lqc].x[oned_vec_index] + q[lqr].x[oned_vec_index];
544 qice = q[lqs].x[oned_vec_index] + q[lqi].x[oned_vec_index] +
545 q[lqg].x[oned_vec_index];

547 e_int =
548 internal_energy(t[oned_vec_index], q[lqv].x[oned_vec_index], qliq,
549 qice, rho[oned_vec_index], dz[oned_vec_index]) +
550 eflx[iv];
551
552 zeta = dt / (2.0 * dz[oned_vec_index]);
553 xrho = std::sqrt(rho_00 / rho[oned_vec_index]);
554
555
556 // simulate how precipitation and its associated energy
557 // move vertically through the atmosphere.
558 for (size_t ix = 0; ix < np; ix++) {
559 if (k >= kmin[iv][qp_ind[ix]]) {
560 vc = vel_scale_factor(qp_ind[ix], xrho, rho[oned_vec_index],
561 t[oned_vec_index],
562 q[qp_ind[ix]].x[oned_vec_index]);
563
564 // simulates precipitation processes, adjusting the vertical distribution of different hydrometeors
565 precip(params[qp_ind[ix]], update, zeta, vc, q[qp_ind[ix]].p[iv],
566 vt[iv][ix], q[qp_ind[ix]].x[oned_vec_index],
567 q[qp_ind[ix]].x[kp1 * ivend + iv], rho[oned_vec_index]);
568 q[qp_ind[ix]].x[oned_vec_index] = update[0];
569 q[qp_ind[ix]].p[iv] = update[1];
570 vt[iv][ix] = update[2];
571 }
572 }

574 pflx[oned_vec_index] = q[lqs].p[iv] + q[lqi].p[iv] + q[lqg].p[iv];
575
576 // updates the flux of energy (eflx) associated with precipitation,
577 eflx[iv] =
578 dt * (q[lqr].p[iv] * (clw * t[oned_vec_index] -
579 cvd * t[kp1 * ivend + iv] - lvc) +
580 pflx[oned_vec_index] * (ci * t[oned_vec_index] -
581 cvd * t[kp1 * ivend + iv] - lsc));
582 pflx[oned_vec_index] = pflx[oned_vec_index] + q[lqr].p[iv];
583 qliq = q[lqc].x[oned_vec_index] + q[lqr].x[oned_vec_index];
584 qice = q[lqs].x[oned_vec_index] + q[lqi].x[oned_vec_index] +
585 q[lqg].x[oned_vec_index];
586 e_int = e_int - eflx[iv];
587
588 // The temperature at each grid point is adjusted based on the updated internal energy
589 t[oned_vec_index] =
590 T_from_internal_energy(e_int, q[lqv].x[oned_vec_index], qliq, qice,
591 rho[oned_vec_index], dz[oned_vec_index]);
592 }
593 }
