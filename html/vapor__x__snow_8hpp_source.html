59TARGET real_t vapor_x_snow(real_t t, real_t p, real_t rho, real_t qs, real_t ns,
60 real_t lambda, real_t eta, real_t ice_dep,
61 real_t dvsw, real_t dvsi, real_t dvsw0, real_t dt) {
62 const real_t a1_vs = static_cast<real_t>(0.4182) * sqrt(graupel_ct::v0s / nu);
63 real_t result = 0.0;
64
65 if (qs > graupel_ct::qmin) {
66 if (t < thermodyn::tmelt) {
67 result = (cnx * ns * eta / rho) * (a0_vs + a1_vs * pow(lambda, a2_vs)) *
68 dvsi / (lambda * lambda + eps);
69
70 // GZ: This limitation, which was missing in the original graupel scheme,
71 // is crucial for numerical stability in the tropics!
72 // a meaningful distiction between cloud ice and snow
73
74 if (result > 0.0)
75 result = fmin(result, dvsi / dt - ice_dep);
76 if (qs <= qs_lim)
77 result = fmin(result, static_cast<real_t>(0.0));
78 } else {
79 if (t > (thermodyn::tmelt - graupel_ct::tx * dvsw0)) {
80 result = (c1_vs / p + c2_vs) * fmin(static_cast<real_t>(0.0), dvsw0) *
81 pow(qs * rho, b_vs);
82 } else {
83 result = (c3_vs + c4_vs * p) * dvsw * pow(qs * rho, b_vs);
84 }
85 }
86 result = fmax(result, -qs / dt);
87 }
88
89 return result;
90}
