50TARGET real_t cloud_to_rain(real_t t, real_t qc, real_t qr, real_t nc) {
51 const real_t au_kernel =
52 x1 / (static_cast<real_t>(20.0) * x2) * (x3 + static_cast<real_t>(2.0)) *
53 (x3 + static_cast<real_t>(4.0)) /
54 pow((x3 + static_cast<real_t>(1.0)), static_cast<real_t>(2.0));
55 real_t result = 0.0;
56
57 if (qc > qmin_ac && t > graupel_ct::tfrz_hom) {
58 real_t tau = fmax(tau_min, fmin(static_cast<real_t>(1.0) - qc / (qc + qr),
59 tau_max)); // time-scale
60 real_t phi = pow(tau, b_phi); // similarity function for autoconversion
61 phi = a_phi * phi *
62 pow((static_cast<real_t>(1.0) - phi), static_cast<real_t>(3.0));
63 real_t xau = au_kernel * pow(qc * qc / nc, static_cast<real_t>(2.)) *
64 (static_cast<real_t>(1.0) +
65 phi / pow(static_cast<real_t>(1.0) - tau,
66 static_cast<real_t>(2.0))); // autoconversion rate
67 real_t xac =
68 ac_kernel * qc * qr *
69 pow((tau / (tau + c_phi)), static_cast<real_t>(4.0)); // accretion rate
70 result = xau + xac;
71 }
72
73 return result;
74}
