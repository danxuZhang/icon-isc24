48TARGET real_t internal_energy(real_t t, real_t qv, real_t qliq, real_t qice,
49 real_t rho, real_t dz) {
50
51 // total water specific mass
52 real_t qtot = qliq + qice + qv;
53
54 // moist isometric specific heat
55 real_t cv = cvd * (static_cast<real_t>(1.0) - qtot) + cvv * qv + clw * qliq +
56 graupel_ct::ci * qice;
57
58 return rho * dz * (cv * t - qliq * graupel_ct::lvc - qice * graupel_ct::lsc);
59}

77TARGET real_t T_from_internal_energy(real_t U, real_t qv, real_t qliq,
78 real_t qice, real_t rho, real_t dz) {
79
80 // total water specific mass
81 real_t qtot = qliq + qice + qv;
82
83 // moist isometric specific heat
84 real_t cv = (cvd * (static_cast<real_t>(1.0) - qtot) + cvv * qv + clw * qliq +
85 graupel_ct::ci * qice) *
86 rho * dz;
87
88 return (U + rho * dz * (qliq * graupel_ct::lvc + qice * graupel_ct::lsc)) /
89 cv;

102[[maybe_unused]] TARGET real_t specific_humidity(real_t pvapor, real_t ptotal) {
103
104 real_t rdv = rd / rv;
105 real_t o_m_rdv = static_cast<real_t>(1.) - rdv;
106
107 return rdv * pvapor / (ptotal - o_m_rdv * pvapor);
108}
