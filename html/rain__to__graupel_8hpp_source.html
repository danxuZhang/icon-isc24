52TARGET real_t rain_to_graupel(real_t t, real_t rho, real_t qc, real_t qr,
53 real_t qi, real_t qs, real_t mi, real_t dvsw,
54 real_t dt) {
55
56 real_t result = 0.0;
57
58 if (qr > graupel_ct::qmin && t < tfrz_rain) {
59 if (t > graupel_ct::tfrz_hom) {
60 if (dvsw + qc <= 0.0 || qr > c4 * qc) {
61 result = (exp(c2 * (tfrz_rain - t)) - c3) * (a1 * pow((qr * rho), b1));
62 }
63 } else {
64 result = qr / dt;
65 }
66 }
67
68 if (fmin(qi, qr) > graupel_ct::qmin &&
69 qs > qs_crit) { // ! rain + ice creating graupel
70 result = result + a2 * (qi / mi) * pow(qr, b2);
71 }
