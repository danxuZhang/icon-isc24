49void io_muphys::input_vector(NcFile &datafile, array_1d_t<real_t> &v,
50 const string input, size_t &ncells, size_t &nlev) {
51 NcVar var;
52 v.resize(ncells * nlev);
53 /* access the input variable */
54 try {
55 var = datafile.getVar(input);
56 } catch (NcNotVar &e) {
57 cout << "FAILURE in accessing " << input << " (no time dimension) *******"
58 << endl;
59 e.what();
60 e.errorCode();
61 }
62 /* read-in input field values */
63 try {
64 array_1d_t<size_t> startp = {0, 0};
65 array_1d_t<size_t> count = {nlev, ncells};
66 var.getVar(startp, count, v.data());
67 } catch (NcNotVar &e) {
68 cout << "FAILURE in reading values from " << input
69 << " (no time dimensions) *******" << endl;
70 e.what();
71 e.errorCode();
72 }
73}

75void io_muphys::input_vector(NcFile &datafile, array_1d_t<real_t> &v,
76 const string input, size_t &ncells, size_t &nlev,
77 size_t itime) {
78 NcVar att = datafile.getVar(input);
79 try {
80 v.resize(ncells * nlev);
81 if (att.isNull()) {
82 throw NC_ERR;
83 }
84 array_1d_t<size_t> startp = {itime, 0, 0};
85 array_1d_t<size_t> count = {1, nlev, ncells};
86 att.getVar(startp, count, v.data());
87 } catch (NcException &e) {
88 e.what();
89 cout << "FAILURE in reading " << input << " **************************"
90 << endl;
91 throw NC_ERR;
92 }

96 const string output, array_1d_t<real_t> &v,
97 size_t &ncells, size_t &nlev,
98 int &deflate_level) {
99 // fortran:column major while c++ is row major
100 NCreal_t ncreal_t;

102

104 var.putVar({i, 0}, {1, ncells}, &v[i * ncells]);
105 }
106 if (deflate_level > 0) {
107 var.setCompression(true, false, deflate_level);
108 }

200static void copy_coordinate_variables(NcFile &datafile, NcFile &inputfile,
201 array_1d_t<std::string> coordinates) {
202 for (auto &coordinate_name : coordinates) {
203 auto coordinate = inputfile.getVar(coordinate_name);
204 /* copy possible new dimensions from input coordinates to the output
205 * befor adding the related data variables */
206 for (NcDim &dim : coordinate.getDims()) {
207 auto currentDims = datafile.getDims();
208 /* map.contains() would be better, but requires c++20 */
209 if (auto search = currentDims.find(dim.getName());
210 search == currentDims.end())
211 datafile.addDim(dim.getName(), dim.getSize());
212 }
213 auto var = datafile.addVar(coordinate.getName(), coordinate.getType(),
214 coordinate.getDims());
215
216 for (auto &[attribute_name, attribute] : coordinate.getAtts()) {
217 std::string dataValues = "default";
218 attribute.getValues(dataValues);
219 var.putAtt(attribute.getName(), attribute.getType(),
220 attribute.getAttLength(), dataValues.c_str());
221 }
222
223 /* Calculate the total size of the varibale */
224 auto dimensions = coordinate.getDims();
225 struct Prod {
226 void operator()(NcDim n) { prod *= n.getSize(); }
227 size_t prod{1};
228 };
229 size_t totalSize =
230 std::for_each(dimensions.cbegin(), dimensions.cend(), Prod()).prod;
231
232 /* Create a one-dimensional vector to store its values */
233 array_1d_t<real_t> oneDimensionalVariable(totalSize);
234
235 /* Copy the original values to the output */
236 coordinate.getVar(oneDimensionalVariable.data());
237 var.putVar(oneDimensionalVariable.data());
238 }
239}
